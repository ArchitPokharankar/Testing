ASP.NET Core MVC - Product Page Project

Here are all the necessary code snippets for the project. Assume your project is named MyWebApp. If you used a different name, remember to change "MyWebApp" in the code below.

1️⃣ Model: The Product Blueprint

File location: Models/Product.cs


namespace MyWebApp.Models
{
    // This is the blueprint for a single product.
    public class Product
    {
        public int Id { get; set; }
        public string Name { get; set; }
        public string Description { get; set; }
        public decimal Price { get; set; }
        public string ImageUrl { get; set; }
    }
}

2️⃣ Controller: The Manager

File location: Controllers/ProductController.cs


using Microsoft.AspNetCore.Mvc;
using MyWebApp.Models; // Tells the controller where to find the Product blueprint
using System.Collections.Generic;
using System.Linq;

namespace MyWebApp.Controllers
{
    public class ProductController : Controller
    {
        // A simple, hardcoded list of our products
        private static List<Product> _products = new List<Product>
        {
            new Product { Id = 1, Name = "Laptop", Description = "A powerful laptop.", Price = 90000, ImageUrl = "/images/laptop.jpg" },
            new Product { Id = 2, Name = "Mobile", Description = "A modern smartphone.", Price = 35000, ImageUrl = "/images/phone.jpg" },
            new Product { Id = 3, Name = "Headphone", Description = "Noise-cancelling headphones.", Price = 7000, ImageUrl = "/images/headphones.jpg" }
        };

        // This action shows the list of ALL products.
        public IActionResult Index()
        {
            // It sends our list of products to the "Index" view.
            return View(_products);
        }

        // This action shows the details for ONE product.
        public IActionResult Details(int id)
        {
            // It finds the product in our list that matches the requested ID.
            var product = _products.FirstOrDefault(p => p.Id == id);

            // It sends that single product to the "Details" view.
            return View(product);
        }
    }
}

3️⃣ View: The Product List Page

File location: Views/Product/Index.cshtml


<!-- This file displays the table of all products -->
@model IEnumerable<MyWebApp.Models.Product>

<h2>Our Products</h2>

<table class="table">
    @foreach (var p in Model)
    {
        <tr>
            <td>@p.Name</td>
            <td>@p.Price.ToString("c")</td> <!-- "c" formats it as currency -->
            <td>
                <!-- This link goes to the "Details" page for the product's Id -->
                <a asp-action="Details" asp-route-id="@p.Id" class="btn btn-primary">View Details</a>
            </td>
        </tr>
    }
</table>
