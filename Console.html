Practical 1 - C# Programs

1️⃣ Simple Calculator

using System;
class Program
{
    static void Main()
    {
        Console.Write("Enter 1st: "); int a = int.Parse(Console.ReadLine());
        Console.Write("Enter 2nd: "); int b = int.Parse(Console.ReadLine());
        Console.Write("Choose (+,-,*,/): "); char op = char.Parse(Console.ReadLine());
        int r = 0;

        switch (op)
        {
            case '+': r = a + b; break;
            case '-': r = a - b; break;
            case '*': r = a * b; break;
            case '/':
                if (b == 0)
                {
                    Console.WriteLine("Invalid division (cannot divide by zero).");
                    return;
                }
                r = a / b;
                break;
            default:
                Console.WriteLine("Invalid operator!");
                return;
        }

        Console.WriteLine($"Result = {r}");
    }
}

2️⃣ Calculator with Functions

using System;
class Program {
    static int Add(int a,int b)=>a+b;
    static int Sub(int a,int b)=>a-b;
    static int Mul(int a,int b)=>a*b;
    static int Div(int a,int b)=>b==0?0:a/b;
    static void Main() {
        Console.Write("Enter 1st: "); int a=int.Parse(Console.ReadLine());
        Console.Write("Enter 2nd: "); int b=int.Parse(Console.ReadLine());
        Console.Write("Choose (+,-,*,/): "); char op=char.Parse(Console.ReadLine());
        if(op=='/' && b==0){Console.WriteLine("Invalid division"); return;}
        int r = op switch {
            '+' => Add(a,b),
            '-' => Sub(a,b),
            '*' => Mul(a,b),
            '/' => Div(a,b),
            _ => 0
        };
        Console.WriteLine($"Result = {r}");
    }
}

4️⃣ Prime Number

using System;
class Program {
    static void Main() {
        Console.Write("Enter number: ");
        int n=int.Parse(Console.ReadLine());
        bool prime = n>1;
        for(int i=2;i<=Math.Sqrt(n);i++)
            if(n%i==0){ prime=false; break; }
        Console.WriteLine(prime ? "Prime" : "Not Prime");
    }
}

6️⃣ Factorial (With Recursion)

using System;
class Program {
    static long Fact(int n)=> (n<=1)?1:n*Fact(n-1);
    static void Main() {
        Console.Write("Enter number: ");
        int n=int.Parse(Console.ReadLine());
        Console.WriteLine(n<0?"Invalid":"Factorial = "+Fact(n));
    }
}

7️⃣ Factorial (Without Recursion)

using System;
class Program {
    static void Main() {
        Console.Write("Enter number: ");
        int n=int.Parse(Console.ReadLine());
        if(n<0){Console.WriteLine("Invalid");return;}
        long f=1; for(int i=2;i<=n;i++) f*=i;
        Console.WriteLine($"Factorial = {f}");
    }
}
