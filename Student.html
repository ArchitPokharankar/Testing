2. Controller (Controllers/StudentController.cs)


// File: Controllers/StudentController.cs

using Microsoft.AspNetCore.Mvc;
using StudentProject.Models; // Use your actual project name
using System.Collections.Generic;
using System.Linq;

namespace StudentProject.Controllers
{
    public class StudentController : Controller
    {
        // This static list acts as a simple in-memory database.
        private static List<Student> students = new List<Student>
        {
            new Student { Id = 1, Name = "Mukesh", Age = 21, Course = "Computer Science" },
            new Student { Id = 2, Name = "Raj", Age = 23, Course = "Bio Technology" }
        };

        // This action shows the list of all students.
        // GET: /Student/ or /Student/Index
        public IActionResult Index()
        {
            // Pass the list of students to the corresponding View.
            return View(students);
        }

        // This action displays the form to add a new student.
        // GET: /Student/Create
        public IActionResult Create()
        {
            return View();
        }

        // This action handles the form submission.
        // POST: /Student/Create
        [HttpPost]
        public IActionResult Create(Student student)
        {
            // A simple way to generate a new unique ID.
            student.Id = students.Any() ? students.Max(s => s.Id) + 1 : 1;
            students.Add(student);
            
            // Redirect back to the Index page to show the updated list.
            return RedirectToAction("Index");
        }
    }
}
